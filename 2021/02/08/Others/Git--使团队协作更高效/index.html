Git初始化

打开控制台, 在D盘创建Project

+| myProject

​ +| .git

​ +| README.md

#更改文件路径
C:\Users\13949>D:
C:\Users\13949\Desktop>md myProject
C:\Users\13949\Desktop>cd myProject
#初始化Git项目, 成功后创建一个文件夹.git
C:\Users\13949\Desktop\myProject>git init
#Initialized empty Git repository in C:/Users/13949/Desktop/myProject/.git/
#新建一个README.md文件
C:\Users\13949\Desktop\myProject>type nul>README.md
#提交README.md至暂存区
C:\Users\13949\Desktop\myProject>git add README.md
#将文件提交到git仓库（-m表示添加本次提交的说明，强制要求写的）
C:\Users\13949\Desktop\myProject>git commit -m "add a README file"
#[master (root-commit) e20f7ab] add a README file
#1 file changed, 0 insertions(+), 0 deletions(-)
#create mode 100644 README.md
#clone项目
C:\Users\13949\Desktop\myProject>git clone https://github.com/hongsheng-xu/xinzhou.github.io
